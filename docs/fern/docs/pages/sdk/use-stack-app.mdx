# useStackApp

A custom React hook for managing a stack of items with loading state.

## Props

This hook doesn't accept any props.

## Example

```tsx
import { useStackApp } from '@stackframe/stack';

function MyComponent() {
  const { items, loading, push, pop } = useStackApp();

  const handlePush = () => {
    push({ id: Date.now(), value: 'New Item' });
  };

  const handlePop = async () => {
    const poppedItem = await pop();
    console.log('Popped item:', poppedItem);
  };

  return (
    <div>
      <button onClick={handlePush}>Push Item</button>
      <button onClick={handlePop} disabled={loading}>
        {loading ? 'Popping...' : 'Pop Item'}
      </button>
      <ul>
        {items.map((item) => (
          <li key={item.id}>{item.value}</li>
        ))}
      </ul>
    </div>
  );
}
```
